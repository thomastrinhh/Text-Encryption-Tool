// Inheritance
class Text: private Encrypt {

    int userChoice;
    
    void setText() {
        cout << "Enter the message you would like to be encrypted: ";
        cin >> text;
    }
    
    void setChoice() {
        cout << "Enter which encryption method you would like to use: ";
        cin >> userChoice;
        
        switch(userChoice) {
            case 1: caesarAlgorithm(); break;
            case 2: vigenereAlgorithm(); break;
            case 3: loganAlgorithm(); break;

        }
